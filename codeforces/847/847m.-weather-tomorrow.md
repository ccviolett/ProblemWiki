```cpp
#include <bits/stdc++.h>

using namespace std;

int n;
int tmp[1000];

int main() {
	cin >> n;
	for (int i = 0; i < n; ++i) 
		cin >> tmp[i];
	if (n == 2) {
		cout << tmp[1] * 2 - tmp[0] << endl;
		return 0;
	}
	for (int i = 2; i < n; ++i) {
		if (tmp[i] - tmp[i - 1] != tmp[i - 1] - tmp[i - 2]) {
			cout << tmp[n - 1] << endl;
			return 0;
		}
	}
	cout << tmp[n - 1] * 2 - tmp[n - 2] << endl;
	return 0;
}
```

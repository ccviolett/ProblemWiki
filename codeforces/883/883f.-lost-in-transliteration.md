```cpp
#include <bits/stdc++.h>

using namespace std;

const int N = 100;

char s[N];
map<string, bool> have;

int main() {
	int n;
	cin >> n;
	int res = 0;
	for (int i = 1; i <= n; ++i) {
		string ts;
		cin >> s;
		int m = strlen(s);
		for (int j = m - 1; j >= 0; --j) {
			if (s[j] == 'u') {
				ts.push_back('o');
				ts.push_back('o');
			} else if (s[j] == 'h') {
				while (j && s[j - 1] == 'k') {
					--j;
				}
				ts.push_back('h');
			} else ts.push_back(s[j]);
		}
		// cout << ts << endl;
		if (!have[ts]) res++;
		have[ts] = true;
	}
	cout << res << endl;
	return 0;
}
```
